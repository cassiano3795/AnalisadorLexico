<program> ::= <statements>

<statements> ::= <statement> <statements>+ | <statement>
<statement> ::= <chardeclaration> | <intdeclaration> | <floatdeclaration> 
			    <ifstatements> | <switchstatement> | <assignstatement>

<chardeclaration> ::= CHAR <ids> SEMI
<intdeclaration> ::= INT <ids> SEMI
<floatdeclaration> ::= FLOAT <ids> SEMI

<assignstatement> ::= <assign> SEMI

<ifstatement> ::= IF LPAREN <condition> RPAREN LBRACKET <statements> RBRACKET <elsestatement>?
<elsestatement> ::= ELSE <ifstatement> | ELSE LBRACKET <statements> RBRACKET

<switchstatement> ::= SWITCH LPAREN ID RPAREN LBRACKET <cases> RBRACKET

<cases> ::= CASE <literal> COLON <statements> <finish> SEMI <cases>* | DEFAULT COLON <statements> <finish> SEMI
<finish> ::= BREAK | <return>

<return> ::= RETURN <operand> | RETURN

<ids> ::= ID | ID COMMA <ids>+ | <assign> <ids>+
<assign> ::= ID ASSIGN <expression> | ID ASSIGN <expression> COMMA <ids>+

<expression> ::= <operand>
		 
<operand> ::= <literal> | <location>

<literal> ::= STRING | INTEGER | FLOATNUMBER
<location> ::= ID
